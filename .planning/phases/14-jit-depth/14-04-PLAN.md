<objective>
Add native conditional branch emission. The thunk must reconstruct x86 EFLAGS from cpu->apsr
just before the branch (research D.15), then emit an x86 jcc rel32 to a per-branch PC-store
sequence. Block terminates at the branch (jit.c is_terminator already includes B_COND/B_UNCOND/
T32_B_COND/T32_BL — see jit.c:24-34), so no further insns emit after a branch.

Pseudocode of the emit at end of block for `B<cond> #imm`:

```
emit_apsr_to_eflags()      ; push/pop with explicit bit assembly
jcc <taken_label>           ; rel32 form: 0F 8X disp32
mov [r15 + PC_OFF], <fallthrough_pc>
jmp <epilogue_check>
taken_label:
mov [r15 + PC_OFF], <taken_pc>
; fall through to epilogue_check (added by codegen_emit)
```

The standard `st_pc(last)` at end of block (added by codegen_emit) becomes the
"epilogue_check" target. Currently codegen_emit unconditionally calls st_pc with the last
ins's PC+size — that's the FALLTHROUGH PC. For terminator branches, st_pc is now redundant
(emit_op already wrote the correct PC via the branch). Skip the trailing st_pc when the last
insn is a terminator branch.

Purpose: JIT-04 success criterion. Most loops in firmware terminate with a B<cond>; without
native B.cond every loop body falls back to interpreter, even though all the body insns are
native-eligible. Closing this unlocks 100M+ IPS on hot loops.

Output: codegen.c gains ~120 lines for branch emission. Test exercises all 14 cond codes
(EQ/NE/CS/CC/MI/PL/VS/VC/HI/LS/GE/LT/GT/LE) by running CMP+B.cond pairs and asserting
final PC matches the expected branch direction.

Step A — emit_apsr_to_eflags. Strategy: pushfq to capture current EFLAGS into rax, clear
the four bits we control (SF/ZF/CF/OF), OR in apsr-derived bits, push back, popfq.

Bit positions in EFLAGS:
- CF = bit 0      (mask 1 << 0  = 0x0001)
- PF = bit 2
- AF = bit 4
- ZF = bit 6      (mask 1 << 6  = 0x0040)
- SF = bit 7      (mask 1 << 7  = 0x0080)
- TF = bit 8
- IF = bit 9
- DF = bit 10
- OF = bit 11     (mask 1 << 11 = 0x0800)
Mask to clear (NZCV positions only): 0x0001 | 0x0040 | 0x0080 | 0x0800 = 0x08C1

```asm
; eax/edx/r10d are volatile in WIN64 — safe to clobber inside thunk after prologue.
; r15 = cpu, r14 = bus.
;
; Read APSR -> ecx:
mov  ecx, [r15 + CG_APSR_OFF]    ; 41 8B 8F disp32       (7B)
; Reserve 8B local:
sub  rsp, 8                       ; 48 83 EC 08            (4B)
pushfq                            ; 9C                      (1B)  rsp -= 8
; Now rsp + 0 holds the original 8 bytes (uninitialized) + pushfq result is at [rsp+0].
; Actually after pushfq: top of stack = EFLAGS qword. Pop it into rax:
pop  rax                          ; 58                      (1B)  rsp += 8
; rax low 64 bits = current EFLAGS. Clear our 4 bits:
and  rax, ~0x08C1ull              ; 48 25 imm32 ... or:
; Simpler with movabs + and (since 0x...FFFFF73E doesn't fit in imm32 sign-extended).
; Use: not 0x08C1 = 0xFFFFFFFFFFFFF73E. 32-bit imm sign-extends to 64-bit if high bit set.
; 0xFFFFF73E sign-extends to 0xFFFFFFFFFFFFF73E — perfect. Encoding:
; and rax, sign_ext_imm32:  48 25 3E F7 FF FF
emit:                              48 25 3E F7 FF FF    (6B)
;
; Build EFLAGS bits from APSR:
;   APSR.N (bit31) -> EFLAGS SF (bit7)
;   APSR.Z (bit30) -> EFLAGS ZF (bit6)
;   APSR.C (bit29) -> EFLAGS CF (bit0)
;   APSR.V (bit28) -> EFLAGS OF (bit11)
; Use BT (bit-test) + SETC + SHL + OR sequence in r10d, then OR into rax.
;
; SF: bt ecx,31; setc r10b; shl r10d, 7; or rax, r10
; ZF: bt ecx,30; setc r10b; shl r10d, 6; or rax, r10
; CF: bt ecx,29; setc r10b;              or rax, r10  (no shift, bit 0)
; OF: bt ecx,28; setc r10b; shl r10d,11; or rax, r10
;
; Encodings:
; bt ecx, imm8         : 0F BA E1 imm8         (4B)
; setc r10b            : 41 0F 92 C2          (4B)  (REX.B=1 for r10b; ModRM 0xC2)
; shl r10d, imm8       : 41 C1 E2 imm8         (4B)
; or rax, r10          : 4C 09 D0              (3B)  (REX.WR=1; ModRM 0xD0)
;
push rax                          ; 50                    (1B)
popfq                             ; 9D                    (1B)
add rsp, 8                        ; 48 83 C4 08           (4B)  balance the leading sub rsp,8
```

WAIT — the leading `sub rsp, 8` is wrong. pushfq pushes 8B then pop pops it. We don't need
extra space. Remove the sub/add. Final sequence:

```asm
mov  ecx, [r15 + CG_APSR_OFF]
pushfq                            ; 9C       rsp -= 8
pop  rax                          ; 58       rax = EFLAGS, rsp += 8
and  rax, 0xFFFFF73E              ; 48 25 3E F7 FF FF  (sign-ext)
; insert NZCV bits (4 occurrences of bt+setc+shl+or):
bt   ecx, 31; setc r10b; shl r10d, 7;  or rax, r10
bt   ecx, 30; setc r10b; shl r10d, 6;  or rax, r10
bt   ecx, 29; setc r10b;               or rax, r10
bt   ecx, 28; setc r10b; shl r10d, 11; or rax, r10
push rax                          ; 50       rsp -= 8
popfq                             ; 9D       rsp += 8 (EFLAGS restored)
```

Stack net change: 0 (pushfq+pop balance, push+popfq balance). Safe under WIN64 alignment
(no offset across call boundary).

Implement helpers in src/core/codegen.c (place after the flag-setter helpers from 14-03):

```c
/* mov ecx, [r15 + CG_APSR_OFF]  (41 8B 8F disp32) */
static void ld_ecx_apsr(codegen_t* cg) {
    emit_b(cg, 0x41); emit_b(cg, 0x8B); emit_b(cg, 0x8F);
    emit_w32(cg, CG_APSR_OFF);
}
/* pushfq (9C); popfq (9D); push rax (50); pop rax (58) */
static void emit_pushfq(codegen_t* cg) { emit_b(cg, 0x9C); }
static void emit_popfq (codegen_t* cg) { emit_b(cg, 0x9D); }
static void emit_push_rax(codegen_t* cg) { emit_b(cg, 0x50); }
static void emit_pop_rax (codegen_t* cg) { emit_b(cg, 0x58); }
/* and rax, sign_ext_imm32  (48 25 imm32) — high bit must be 1 for sign extension */
static void and_rax_imm32(codegen_t* cg, u32 v) {
    emit_b(cg, 0x48); emit_b(cg, 0x25); emit_w32(cg, v);
}
/* bt ecx, imm8  (0F BA E1 imm8) */
static void bt_ecx(codegen_t* cg, u8 n) {
    emit_b(cg, 0x0F); emit_b(cg, 0xBA); emit_b(cg, 0xE1); emit_b(cg, n);
}
/* setc r10b  (41 0F 92 C2) */
static void setc_r10b(codegen_t* cg) {
    emit_b(cg, 0x41); emit_b(cg, 0x0F); emit_b(cg, 0x92); emit_b(cg, 0xC2);
}
/* or rax, r10  (4C 09 D0) — REX.W+REX.R, opcode 09, ModRM mod=11 reg=010(r10) rm=000(rax) */
static void or_rax_r10(codegen_t* cg) {
    emit_b(cg, 0x4C); emit_b(cg, 0x09); emit_b(cg, 0xD0);
}

static void emit_apsr_to_eflags(codegen_t* cg) {
    ld_ecx_apsr(cg);
    emit_pushfq(cg);
    emit_pop_rax(cg);
    /* Clear SF, ZF, CF, OF in rax: mask 0xFFFFFFFFFFFFF73E (sign-ext from 0xFFFFF73E) */
    and_rax_imm32(cg, 0xFFFFF73Eu);
    /* Insert N -> SF (bit 7) */
    bt_ecx(cg, 31u); setc_r10b(cg); shl_r10d(cg, 7u);  or_rax_r10(cg);
    /* Insert Z -> ZF (bit 6) */
    bt_ecx(cg, 30u); setc_r10b(cg); shl_r10d(cg, 6u);  or_rax_r10(cg);
    /* Insert C -> CF (bit 0) — no shift needed */
    bt_ecx(cg, 29u); setc_r10b(cg);                    or_rax_r10(cg);
    /* Insert V -> OF (bit 11) */
    bt_ecx(cg, 28u); setc_r10b(cg); shl_r10d(cg, 11u); or_rax_r10(cg);
    emit_push_rax(cg);
    emit_popfq(cg);
}
```

Note: setc_r10b zeroes the high bytes of r10 (movzx-like behavior of setcc on a byte
register? No — setcc only writes to the byte; high bytes are untouched). Use SETC + the
fact that we always XOR-clear r10 first... actually SETC writes a byte; previous bits 8..63
of r10 may still hold dirty data from the prior bt+setc+shl. The shl r10d,N preserves only
the byte we just set IFF the high bytes were already zero from a prior op. But across the
4 iterations they are NOT cleared.

Fix: prepend each iteration with `xor r10, r10` (3B: 4D 31 D2). Or use MOVZX directly:
`movzx r10d, sil` — no, sil is rsi's low byte and we use rsi/esi for accumulator. Cleanest:

```c
/* Replace setc_r10b with: xor r10d,r10d; setc r10b   (no need for rzero of upper) */
static void setc_r10d(codegen_t* cg) {
    /* xor r10d, r10d  (45 31 D2)  — zero r10 fully */
    emit_b(cg, 0x45); emit_b(cg, 0x31); emit_b(cg, 0xD2);
    /* setc r10b */
    emit_b(cg, 0x41); emit_b(cg, 0x0F); emit_b(cg, 0x92); emit_b(cg, 0xC2);
}
```

Replace all 4 usages above with setc_r10d. Total cost: +3B per use = +12B; total
emit_apsr_to_eflags ~= 60B. Acceptable.

CRITICAL: The pushfq/popfq and bt/setc operations clobber EFLAGS internally. That is FINE —
we are about to issue a jcc that reads EFLAGS, and our LAST instruction before the jcc is
popfq which RESTORES the EFLAGS we built. Anything between popfq and jcc must not modify
SF/ZF/CF/OF. We emit jcc immediately after popfq — safe.

Step B — emit_b_cond. ARM cond -> x86 jcc rel32 opcode (research D.14). Emit the mapping
table at compile time:

```c
/* ARM cond -> x86 jcc near (rel32) opcode 2nd byte (after 0F prefix). */
static u8 cond_to_jcc_rel32(u8 cond) {
    switch (cond) {
        case 0x0: return 0x84;   /* je   (EQ) */
        case 0x1: return 0x85;   /* jne  (NE) */
        case 0x2: return 0x83;   /* jae  (CS/HS) */
        case 0x3: return 0x82;   /* jb   (CC/LO) */
        case 0x4: return 0x88;   /* js   (MI) */
        case 0x5: return 0x89;   /* jns  (PL) */
        case 0x6: return 0x80;   /* jo   (VS) */
        case 0x7: return 0x81;   /* jno  (VC) */
        case 0x8: return 0x87;   /* ja   (HI) */
        case 0x9: return 0x86;   /* jbe  (LS) */
        case 0xA: return 0x8D;   /* jge  (GE) */
        case 0xB: return 0x8C;   /* jl   (LT) */
        case 0xC: return 0x8F;   /* jg   (GT) */
        case 0xD: return 0x8E;   /* jle  (LE) */
        default:  return 0x00;   /* AL, NV — handled elsewhere as unconditional / unreachable */
    }
}

/* Emit B<cond> #imm: terminates block. Lays out as:
   apsr_to_eflags;
   jcc rel32 -> taken_label;
   st_pc fallthrough_pc;
   jmp short +N (skip taken store);     <-- 14B for st_pc + 2B jmp = goto label
   taken_label:
   st_pc taken_pc;
   (epilogue_check follows; codegen_emit will NOT add trailing st_pc when last is a terminator) */
static void emit_b_cond(codegen_t* cg, const insn_t* i) {
    if (i->cond == 0xE) {
        /* AL = always; treat as unconditional B */
        u32 target = i->pc + 4u + i->imm;
        st_pc(cg, target);
        return;
    }
    if (i->cond == 0xF) {
        /* NV = never; treat as fallthrough */
        u32 fall = i->pc + i->size;
        st_pc(cg, fall);
        return;
    }

    emit_apsr_to_eflags(cg);

    u32 target_pc = i->pc + 4u + i->imm;
    u32 fall_pc   = i->pc + i->size;

    /* PRECONDITION: i->imm is the SIGNED BYTE OFFSET. The decoder pre-applies the <<1
   shift for both T1 OP_B_COND (imm8 << 1) and T32 OP_T32_B_COND (imm20 << 1). The
   formula `target_pc = i->pc + 4 + i->imm` is correct ONLY because i->imm is already
   shifted. Verify by reading executor.c handling for OP_B_COND / OP_T32_B_COND.
   If the decoder is later refactored to store the raw unshifted imm, every native
   B.cond will be off by 2x. */

    /* jcc near rel32: 0F 8X disp32. We need disp32 that lands at "taken_label",
       which is AFTER the fallthrough store (11B) and jmp_short (2B) = 13B further
       from the END of this jcc instruction (jcc itself is 6B). So disp32 = 13. */
    emit_b(cg, 0x0F);
    emit_b(cg, cond_to_jcc_rel32(i->cond));
    emit_w32(cg, 13u);            /* disp32: jump 13 bytes forward (past fallthrough store + jmp short) */

    /* fallthrough: st_pc fall_pc */
    st_pc(cg, fall_pc);            /* 11B: 41 C7 87 disp32 imm32 */

    /* jmp short over the taken store. jmp short rel8: EB NN. NN = bytes to skip = 11 (taken store size). */
    emit_b(cg, 0xEB);
    emit_b(cg, 11u);

    /* taken_label: st_pc target_pc */
    st_pc(cg, target_pc);          /* 11B */
}
```

Verify byte count:
- emit_apsr_to_eflags: ~60B
- jcc rel32: 6B
- fallthrough st_pc: 11B
- jmp short rel8: 2B
- taken st_pc: 11B
- TOTAL B.cond emit: ~90B per branch.

The 13B disp32 lands exactly at the "taken store" because: from end of jcc (6B inclusive)
forward 13B passes the 11B fallthrough st_pc and 2B jmp short, lands at the taken st_pc.
Correct.

```c

#include "core/codegen.h"

static jit_t s_jit;

extern bool execute(cpu_t* c, bus_t* bus, const insn_t* i);

/* Replicate cond_pass from executor.c:18-41 for verification truth: */
static bool cond_truth(u32 apsr, u8 cond) {
    bool n = (apsr >> 31) & 1u;
    bool z = (apsr >> 30) & 1u;
    bool c = (apsr >> 29) & 1u;
    bool v = (apsr >> 28) & 1u;
    switch (cond) {
        case 0x0: return z;
        case 0x1: return !z;
        case 0x2: return c;
        case 0x3: return !c;
        case 0x4: return n;
        case 0x5: return !n;
        case 0x6: return v;
        case 0x7: return !v;
        case 0x8: return c && !z;
        case 0x9: return !c || z;
        case 0xA: return n == v;
        case 0xB: return n != v;
        case 0xC: return !z && (n == v);
        case 0xD: return z || (n != v);
        default:  return true;
    }
}

static void run_branch(u32 a, u32 b, u8 cond, u32 fall_imm /*B.cond i->imm*/, const char* tag) {
    bus_t bus; bus_init(&bus);
    bus_add_flat(&bus, "flash", 0x00000000u, 4096u, false);
    bus_add_flat(&bus, "sram",  0x20000000u, 4096u, true);

    cpu_t c = {0};
    cpu_reset(&c, CORE_M4);
    c.r[1] = a; c.r[2] = b;
    c.r[REG_PC] = 0u;

    /* Build [CMP r1,r2; B.cond +imm] */
    insn_t cmp_ins = {0};
    cmp_ins.op = OP_CMP_REG; cmp_ins.rn = 1u; cmp_ins.rm = 2u;
    cmp_ins.pc = 0u; cmp_ins.size = 2u;

    insn_t b_ins = {0};
    b_ins.op = OP_B_COND; b_ins.cond = cond;
    b_ins.imm = fall_imm; b_ins.pc = 2u; b_ins.size = 2u;
    /* B.cond target = 2 + 4 + fall_imm = 6 + fall_imm */
    /* fallthrough = 2 + 2 = 4 */

    insn_t pair[2] = {cmp_ins, b_ins};

    memset(&s_jit, 0, sizeof s_jit);
    jit_init(&s_jit);
    cg_thunk_t fn = codegen_emit(&s_jit.cg, pair, 2u);
    TEST(fn != NULL, tag);
    TEST(fn(&c, &bus), tag);

    /* Compute expected PC by interpreter logic */
    u32 expected_pc;
    if (cond_truth(c.apsr, cond))
        expected_pc = 2u + 4u + fall_imm;     /* taken: target = pc(of B) + 4 + imm */
    else
        expected_pc = 2u + 2u;                /* fallthrough = pc(of B) + size */

    if (c.r[REG_PC] != expected_pc) {
        fprintf(stderr, "BRANCH MISMATCH %s cond=%x a=%08x b=%08x apsr=%08x: native PC=%08x expected=%08x\n",
                tag, cond, a, b, c.apsr, c.r[REG_PC], expected_pc);
    }
    TEST(c.r[REG_PC] == expected_pc, tag);
}

    u32 imm = 0x10u;

    run_branch(0u, 1u, 0xA, imm, "GE fall");

    /* LT: N!=V (a < b signed) */
    run_branch(0u, 1u, 0xB, imm, "LT taken");
    run_branch(10u, 5u, 0xB, imm, "LT fall");

    /* GT: !Z && N==V */
    run_branch(10u, 5u, 0xC, imm, "GT taken");
    run_branch(5u, 5u, 0xC, imm, "GT fall (eq)");
    run_branch(0u, 1u, 0xC, imm, "GT fall (lt)");

    /* LE: Z || N!=V */
    run_branch(5u, 5u, 0xD, imm, "LE taken (eq)");
    run_branch(0u, 1u, 0xD, imm, "LE taken (lt)");
    run_branch(10u, 5u, 0xD, imm, "LE fall");

    /* B_UNCOND: PC = pc+4+imm always */

        bus_t bus; bus_init(&bus);
        bus_add_flat(&bus, "flash", 0x00000000u, 4096u, false);
        bus_add_flat(&bus, "sram",  0x20000000u, 4096u, true);
        cpu_t c = {0}; cpu_reset(&c, CORE_M4); c.r[REG_PC] = 0u;

        memset(&s_jit, 0, sizeof s_jit); jit_init(&s_jit);
        cg_thunk_t fn = codegen_emit(&s_jit.cg, &b, 1u);

        TEST(c.r[REG_PC] == 0u + 4u + 0x20u, "B_UNCOND PC wrong");
```
