<objective>
Fix the WIN64 ABI bug in the JIT thunk emitter. Every native thunk must save the WIN64 arg
registers (rcx=cpu, rdx=bus) into non-volatile r15/r14 at prologue, allocate 32B shadow space,
and access cpu_t via [r15+offset] consistently. This is foundational: without the prologue/
shadow space, helper-call thunks (LDR/STR in 14-02) corrupt rdi when they set up rcx for
bus_read, and any function call inside a thunk crashes due to misaligned stack.

Also export jit_flush(jit_t*) for full TB-cache invalidation, and wire snap_restore to call
it (TT safety: after snap_restore the JIT cache may reference dead PCs if firmware changed).

Purpose: JIT-02..JIT-04 all require helper-call or jcc emission inside thunks. The current
SYSV-style codegen (rdi=cpu, rsi=bus) only works on Windows by accident — jit_run leaves rdi
holding cpu_ptr at the call site (objdump-confirmed: jit_run+0xC3: mov %rdi,%rcx). Any future
helper-call thunk that emits "call bus_read" via WIN64 ABI will clobber rdi with the helper's
arg setup and break all subsequent [rdi+...] accesses. Fix once now; all downstream plans
emit r15-relative addressing.

Output: codegen.c uses [r15+...] throughout; thunks have WIN64-correct prologue/epilogue;
jit_flush exported and wired into snap_restore; smoke test proves the round-trip works.
</objective>

Replace the ld/st helpers (current lines 24-43) with r15-relative versions. Encoding notes:
`mov eax, [r15 + disp32]` = `41 8B 87 disp32` (REX.B=1, opcode 8B, ModRM 0x87 = mod=10 disp32,
reg=000=eax, rm=111 with REX.B=1 -> r15). Same for ecx (reg=001 -> ModRM 0x8F).

```c
/* mov eax, [r15 + CG_R_OFF + r*4]  ->  41 8B 87 disp32 */
static void ld_eax(codegen_t* cg, u8 r) {
    emit_b(cg, 0x41); emit_b(cg, 0x8B); emit_b(cg, 0x87);
    emit_w32(cg, CG_R_OFF + (u32)r * 4u);
}
/* mov ecx, [r15 + CG_R_OFF + r*4]  ->  41 8B 8F disp32 */
static void ld_ecx(codegen_t* cg, u8 r) {
    emit_b(cg, 0x41); emit_b(cg, 0x8B); emit_b(cg, 0x8F);
    emit_w32(cg, CG_R_OFF + (u32)r * 4u);
}
/* mov [r15 + CG_R_OFF + r*4], eax  ->  41 89 87 disp32 */
static void st_eax(codegen_t* cg, u8 r) {
    emit_b(cg, 0x41); emit_b(cg, 0x89); emit_b(cg, 0x87);
    emit_w32(cg, CG_R_OFF + (u32)r * 4u);
}
/* mov dword [r15 + CG_PC_OFF], imm32  ->  41 C7 87 disp32 imm32 */
static void st_pc(codegen_t* cg, u32 pc) {
    emit_b(cg, 0x41); emit_b(cg, 0xC7); emit_b(cg, 0x87);
    emit_w32(cg, CG_PC_OFF);
    emit_w32(cg, pc);
}
```

Step C — add prologue/epilogue emitters. Pitfall 6 in research: WIN64 requires 16B stack
alignment at every call site. Entry rsp is 8 mod 16 (because the ret address is on stack);
4 pushes (32B = 0 mod 16) restores 16B alignment, then `sub rsp,32` keeps it aligned and
provides the mandatory 32B shadow space for any future helper call.

Encoding bytes:
- `push r15` = `41 57`
- `push r14` = `41 56`
- `push rbx` = `53`
- `push rsi` = `56`
- `sub rsp, 32` = `48 83 EC 20`
- `mov r15, rcx` = `49 89 CF`
- `mov r14, rdx` = `49 89 D6`
- `add rsp, 32` = `48 83 C4 20`
- `pop rsi` = `5E`
- `pop rbx` = `5B`
- `pop r14` = `41 5E`
- `pop r15` = `41 5F`

```c
static void emit_prologue(codegen_t* cg) {
    emit_b(cg, 0x41); emit_b(cg, 0x57);             /* push r15 */
    emit_b(cg, 0x41); emit_b(cg, 0x56);             /* push r14 */
    emit_b(cg, 0x53);                                /* push rbx */
    emit_b(cg, 0x56);                                /* push rsi */
    emit_b(cg, 0x48); emit_b(cg, 0x83); emit_b(cg, 0xEC); emit_b(cg, 0x20);  /* sub rsp,32 */
    emit_b(cg, 0x49); emit_b(cg, 0x89); emit_b(cg, 0xCF);                    /* mov r15,rcx */
    emit_b(cg, 0x49); emit_b(cg, 0x89); emit_b(cg, 0xD6);                    /* mov r14,rdx */
}

static void emit_epilogue_ok(codegen_t* cg) {
    emit_b(cg, 0x48); emit_b(cg, 0x83); emit_b(cg, 0xC4); emit_b(cg, 0x20);  /* add rsp,32 */
    emit_b(cg, 0x5E);                                /* pop rsi */
    emit_b(cg, 0x5B);                                /* pop rbx */
    emit_b(cg, 0x41); emit_b(cg, 0x5E);             /* pop r14 */
    emit_b(cg, 0x41); emit_b(cg, 0x5F);             /* pop r15 */
    emit_b(cg, 0xB0); emit_b(cg, 0x01);             /* mov al, 1 */
    emit_b(cg, 0xC3);                                /* ret */
}
```

Replace `ret_true(cg)` (current lines 54-57) with `emit_epilogue_ok(cg)`. Delete `ret_true`.

Step B — implement in src/core/jit.c, append after jit_reset_counters (current line 101-104).

CRITICAL: `lookup_n` AND `n_blocks` must BOTH be reset to 0. Without `lookup_n = 0`, every
post-flush install() appends at the stale lookup_n position; lookups still scan the old
lookup_n slots (all of which now hold lookup_idx[i] = -1) producing silent misses. Without
`n_blocks = 0`, the cache is not actually empty — next compile_block writes into blocks[N+1]
instead of blocks[0]. The struct in jit.h has both fields (n_blocks at line 37, lookup_n at
line 40); the body must zero both:

```c
void jit_flush(jit_t* g) {
    if (!g) return;
    g->n_blocks     = 0u;
    g->lookup_n     = 0;            /* CRITICAL: without this, install() appends at stale pos and lookup scan misses */
    g->jit_steps    = 0u;
    g->native_steps = 0u;
    g->interp_steps = 0u;
    g->cg.used      = 0u;
    for (u32 i = 0; i < JIT_MAX_BLOCKS; ++i) {
        g->lookup_idx[i] = -1;
        g->lookup_pc[i]  = 0u;
        g->counters[i]   = 0u;
    }
}
```

The `cg.used = 0` reset reclaims the entire codegen buffer. Block pointers in blocks[].native
become stale (point into the now-freed buffer region), but they are unreachable: lookup_idx
all -1, lookup_n=0, n_blocks=0. Next codegen_emit overwrites the buffer from offset 0.

Step C — wire snap_restore. In src/core/tt.c around line 135 (current
`jit_reset_counters(g_jit_for_tt);`), expand to flush the entire TB cache. Replace the
single-line call with:

```c
    /* TT safety: after snap_restore, all compiled TBs reference the pre-restore
       PC layout. Flush full cache; TBs will recompile lazily as PCs heat up. */

```

This subsumes jit_reset_counters (jit_flush zeroes counters too). Remove the now-redundant
jit_reset_counters call.

Cost analysis: jit_flush zeroes ~32KB of jit_t (1024 * 12B per slot) -> ~10us memcpy-equivalent.
Plus 2MB cg.buffer index reset (0 cost; just sets used=0). Total <100us. Adds <1ms to rewind
latency; TT-06 budget is 100ms; well within bounds (research H.29 confirmed).

```c
#include "test_harness.h"
#include "core/cpu.h"
#include "core/bus.h"
#include "core/jit.h"
#include "core/codegen.h"
#include "core/decoder.h"
#include <string.h>

static jit_t s_jit;

/* Hot-threshold short-circuit: feed JIT_HOT_THRESHOLD probes at the same PC,
   then a real run that triggers compile_block + native invocation. */

extern bool execute(cpu_t* c, bus_t* bus, const insn_t* i);

int main(void) {
    bus_t bus; bus_init(&bus);
    bus_add_flat(&bus, "flash", 0x00000000u, 4096u, false);
    bus_add_flat(&bus, "sram",  0x20000000u, 4096u, true);

    /* Encode a minimal Thumb-1 stream at PC=0:
       MOV.W r0, #0xDEADBEEF won't fit; use a sequence of MOVs/ORRs that
       collectively load 0xDEADBEEF into r0. Simpler: hand-build an insn_t
       of OP_T32_MOVW with imm=0xDEADBEEF and feed codegen_emit directly. */
    cpu_t cpu = {0};
    cpu_reset(&cpu, CORE_M4);
    cpu.r[REG_PC] = 0u;

    /* Build a 1-insn block via codegen path (bypasses decoder). */
    memset(&s_jit, 0, sizeof s_jit);
    jit_init(&s_jit);

    insn_t ins = {0};
    ins.op = OP_T32_MOVW;
    ins.rd = 0u;
    ins.imm = 0xDEADBEEFu;
    ins.pc = 0u;
    ins.size = 4u;          /* T32 = 4 bytes */

    cg_thunk_t fn = codegen_emit(&s_jit.cg, &ins, 1u);
    TEST(fn != NULL, "codegen_emit returned NULL");

    /* Direct invocation: bypass jit_run wrapping. WIN64 will pass &cpu in rcx,
       &bus in rdx; the prologue saves them into r15/r14. */
    bool ok = fn(&cpu, &bus);
    TEST(ok, "thunk returned false");
    TEST(cpu.r[0] == 0xDEADBEEFu, "r[0] not loaded by thunk");
    TEST(cpu.r[REG_PC] == ins.pc + ins.size, "PC not advanced by thunk");

    /* Test 2: jit_flush wipes everything */
    s_jit.n_blocks = 7u;       /* fake state */
    s_jit.lookup_n = 13;       /* fake stale lookup count */
    s_jit.cg.used  = 12345u;
    s_jit.counters[42] = 99u;
    s_jit.lookup_idx[10] = 5;
    s_jit.lookup_pc[10] = 0xCAFEBABEu;

    jit_flush(&s_jit);
    TEST(s_jit.n_blocks == 0u, "n_blocks not zeroed");
    TEST(s_jit.lookup_n == 0,  "lookup_n not zeroed");
    TEST(s_jit.cg.used == 0u, "cg.used not zeroed");
    TEST(s_jit.counters[42] == 0u, "counter not zeroed");
    TEST(s_jit.lookup_idx[10] == -1, "lookup_idx not reset to -1");
    TEST(s_jit.lookup_pc[10] == 0u, "lookup_pc not zeroed");

    /* Test 3: a 2-insn block with two MOV_IMMs, verify both regs end up correct */
    cpu.r[0] = 0u; cpu.r[1] = 0u; cpu.r[REG_PC] = 0u;
    insn_t a = {0}, b = {0};
    a.op = OP_T32_MOVW; a.rd = 0u; a.imm = 0x11112222u; a.pc = 0u; a.size = 4u;
    b.op = OP_T32_MOVW; b.rd = 1u; b.imm = 0x33334444u; b.pc = 4u; b.size = 4u;
    insn_t pair[2] = {a, b};
    cg_thunk_t fn2 = codegen_emit(&s_jit.cg, pair, 2u);
    TEST(fn2 != NULL, "codegen_emit pair returned NULL");
    TEST(fn2(&cpu, &bus), "pair thunk returned false");
    TEST(cpu.r[0] == 0x11112222u, "r0 wrong after pair");
    TEST(cpu.r[1] == 0x33334444u, "r1 wrong after pair");
    TEST(cpu.r[REG_PC] == 8u, "PC not 8 after 2 T32 insns");

    PASS();
}
```
