    double ips_m = 0.0;
#endif

    if (g) gdb_close(g);

    fprintf(stderr, "halted after %llu instructions\n", (unsigned long long)n);

    fprintf(stderr, "IPS: %.2fM  elapsed: %.1fms\n", ips_m, elapsed_s * 1000.0);

    fprintf(stderr, "R0=%08x R1=%08x R2=%08x R3=%08x\n",

Test layout:

```c
#include "test_harness.h"
#include "core/cpu.h"
#include "core/bus.h"
#include "core/jit.h"
#include "core/run.h"
#include "periph/systick.h"
#include "periph/scb.h"
#include "periph/mpu.h"
#include "periph/dwt.h"
#include "periph/stm32.h"
#include "periph/eth.h"
#include "periph/uart.h"
#include "core/nvic.h"
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#ifdef _WIN32
  #include <windows.h>
#endif

#define FLASH_BASE 0x00000000u
#define FLASH_SIZE (1u << 20)
#define SRAM_BASE  0x20000000u
#define SRAM_SIZE  (256u << 10)

static jit_t s_jit;

extern dwt_t* g_dwt_for_run;
extern nvic_t* g_nvic_for_run;
extern cpu_t* g_cpu_for_scb;

static u8* read_file(const char* p, u32* sz) {
    FILE* f = fopen(p, "rb");
    if (!f) return NULL;
    fseek(f, 0, SEEK_END);
    long n = ftell(f);
    fseek(f, 0, SEEK_SET);
    if (n <= 0) { fclose(f); return NULL; }
    u8* b = (u8*)malloc((size_t)n);
    if (!b) { fclose(f); return NULL; }
    if (fread(b, 1, (size_t)n, f) != (size_t)n) { free(b); fclose(f); return NULL; }
    fclose(f);
    *sz = (u32)n;
    return b;
}

static u8* find_test7(u32* sz) {
    const char* env = getenv("LECERF_FIRMWARE_DIR");
    char path[512];
    if (env) {
        snprintf(path, sizeof path, "%s/test7_freertos/test7_freertos.bin", env);
        u8* b = read_file(path, sz); if (b) return b;
    }
    static const char* roots[] = {
        "firmware/test7_freertos/test7_freertos.bin",
        "../firmware/test7_freertos/test7_freertos.bin",
        "../../firmware/test7_freertos/test7_freertos.bin",
    };
    for (size_t k = 0; k < sizeof(roots)/sizeof(roots[0]); ++k) {
        u8* b = read_file(roots[k], sz); if (b) return b;
    }
    return NULL;
}

int main(void) {
    if (getenv("LECERF_BENCH_SKIP")) {
        fprintf(stderr, "bench skipped via LECERF_BENCH_SKIP\n");
        PASS();
    }

    u32 sz = 0;
    u8* blob = find_test7(&sz);
    TEST(blob != NULL, "test7_freertos.bin not found");
    TEST(sz > 0u && sz <= FLASH_SIZE, "blob size out of range");

    bus_t bus; bus_init(&bus);
    bus_add_flat(&bus, "flash", FLASH_BASE, FLASH_SIZE, false);
    bus_add_flat(&bus, "sram",  SRAM_BASE,  SRAM_SIZE,  true);
    static uart_t u0 = {0};   uart_attach(&bus, &u0);
    static systick_t st = {0}; systick_attach(&bus, &st);
    static scb_t scb = {0};   scb_attach(&bus, &scb);
    static mpu_t mpu = {0};   mpu_attach(&bus, &mpu);
    static stm32_t s = {0};   stm32_attach(&bus, &s); s.quiet = true;
    static dwt_t dwt = {0};   dwt_attach(&bus, &dwt); g_dwt_for_run = &dwt;
    static nvic_t nv = {0};   nvic_attach(&bus, &nv); g_nvic_for_run = &nv;
    static eth_t eth = {0};   eth_attach(&bus, &eth);
    bus_load_blob(&bus, FLASH_BASE, blob, sz);
    free(blob);

    cpu_t cpu;
    cpu_reset(&cpu, CORE_M4);
    g_cpu_for_scb = &cpu;
    cpu.msp = bus_r32(&bus, 0x0u);
    cpu.r[REG_SP] = cpu.msp;
    cpu.r[REG_PC] = bus_r32(&bus, 0x4u) & ~1u;

    memset(&s_jit, 0, sizeof s_jit);
    jit_init(&s_jit);

    /* Warmup: run a few thousand cycles so blocks compile, JIT promotes hot. */
    (void)run_steps_full_g(&cpu, &bus, 50000ull, &st, &scb, &s_jit);

    /* Reset to known position via a fresh load (or accept warmup as part of timing).
       Per JIT-06: 5M instructions <50ms. test7 halts at ~3.05M naturally; with the
       warmup, we're effectively measuring the steady-state hot loop. */

    /* Zero peripherals to remove warmup-induced state (pending IRQ flags, dirty counters,
       stale UART rx_q, etc.). Without this, SysTick may fire an IRQ into the timed run
       and add non-deterministic latency that makes the <50ms assertion flaky. */
    memset(&st,  0, sizeof st);
    memset(&scb, 0, sizeof scb);
    memset(&nv,  0, sizeof nv);
    memset(&mpu, 0, sizeof mpu);
    memset(&dwt, 0, sizeof dwt);
    memset(&s,   0, sizeof s);    /* stm32_p */
    memset(&u0,  0, sizeof u0);   /* uart */
    memset(&eth, 0, sizeof eth);
    /* re-init via *_init() if available (mirrors tools/main.c reset path) */

    /* Re-init for the timed run: */
    bus_init(&bus);
    bus_add_flat(&bus, "flash", FLASH_BASE, FLASH_SIZE, false);
    bus_add_flat(&bus, "sram",  SRAM_BASE,  SRAM_SIZE,  true);
    /* Re-attach (peripherals are stateless wrt the cpu blob; bus_init wipes regs[]) */
    uart_attach(&bus, &u0);
    systick_attach(&bus, &st);
    scb_attach(&bus, &scb);
    mpu_attach(&bus, &mpu);
    stm32_attach(&bus, &s);
    dwt_attach(&bus, &dwt);
    nvic_attach(&bus, &nv);
    eth_attach(&bus, &eth);
    /* Reload the blob */
    u32 sz2 = 0;
    u8* blob2 = find_test7(&sz2);
    TEST(blob2 != NULL && sz2 == sz, "reload mismatch");
    bus_load_blob(&bus, FLASH_BASE, blob2, sz2);
    free(blob2);

    cpu_reset(&cpu, CORE_M4);
    cpu.msp = bus_r32(&bus, 0x0u);
    cpu.r[REG_SP] = cpu.msp;
    cpu.r[REG_PC] = bus_r32(&bus, 0x4u) & ~1u;

    /* Timed 5M-step run */
#ifdef _WIN32
    LARGE_INTEGER freq, t0, t1;
    QueryPerformanceFrequency(&freq);
    QueryPerformanceCounter(&t0);
#endif
    u64 n = run_steps_full_g(&cpu, &bus, 5000000ull, &st, &scb, &s_jit);
#ifdef _WIN32
    QueryPerformanceCounter(&t1);
    double elapsed_s = (double)(t1.QuadPart - t0.QuadPart) / (double)freq.QuadPart;
    double ips_m = (elapsed_s > 0.0) ? ((double)n / elapsed_s / 1e6) : 0.0;
    fprintf(stderr, "[bench] insns=%llu IPS=%.2fM elapsed=%.1fms\n",
            (unsigned long long)n, ips_m, elapsed_s * 1000.0);
    TEST(elapsed_s < 0.050, "5M-cap test7 took >= 50ms (slowdown regression)");
    TEST(ips_m >= 100.0, "IPS dropped below 100M");
#else
    fprintf(stderr, "[bench] non-Windows: skipping timing assertion\n");
#endif

    TEST(n > 0u, "no instructions ran");

    PASS();
}
```

## Key decisions

- **WIN64 ABI fix first**: research identified [rdi+...] thunks worked only by accident on
  Windows (jit_run leaves rdi=cpu_ptr). Any helper-call thunk would corrupt rdi. Fixed once
  in 14-01: prologue saves rcx -> r15 (cpu) and rdx -> r14 (bus); all addressing uses r15.
- **Pseudo-chain over patched-chain**: 14-05 uses a tight while-loop in jit_run_chained
  rather than QEMU-style REL32 patching. Avoids back-reference list and invalidation
  complexity; adequate for 100M+ IPS target. True patched chaining deferred (Phase 15+).
- **Generation reset over per-slot LRU**: 14-05 wipes the entire cache on n_blocks overflow.
  Simpler than QEMU's slot-LRU; FreeRTOS steady-state ~200 unique TBs vs 1024 cap, so eviction
  rare in practice.
- **TT safety**: snap_restore in tt.c calls jit_flush (added in 14-01); chained loop respects
  max_steps so run_until_cycle stays cycle-precise (overshoot bounded by JIT_MAX_BLOCK_LEN-1
  = 31 cycles).
- **Latent ADD/SUB native flag bug fixed**: research C.10 noted T1 ADD_REG/SUB_REG/IMM3/IMM8
  in pre-Phase-14 codegen DID NOT update APSR. Masked because CMP/B.cond fell back to
  interpreter (which corrected flags). Fixed in 14-03: T1 ops always emit flag update.

## Open items deferred to Phase 15+

- Patch-on-first-use chaining (QEMU-style REL32 patching) — adds ~5% IPS; defer
- Direct SRAM access optimization (skip bus_read for known-flat regions) — adds ~5x per LDR;
  baked-pointer invalidation is complex; defer
- LDM/STM native (variable count, complex writeback) — interpreter handles at acceptable cost
- IT block native (would require itstate tracking inside thunk) — IT is rare in FreeRTOS hot

- Per-slot LRU eviction (vs current generation reset) — only needed if firmware has >1024
  unique hot TBs; FreeRTOS does not
