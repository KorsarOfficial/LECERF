Create docker/Dockerfile (~40 lines):

  # Stage 1: builder -- compiles liblecerf.so for musl + builds the wheel
  FROM python:3.12-alpine AS builder
  RUN apk add --no-cache --virtual .build-deps \
          gcc musl-dev cmake make linux-headers
  WORKDIR /src
  COPY CMakeLists.txt /src/
  COPY include/ /src/include/
  COPY src/ /src/src/
  COPY python/ /src/python/
  RUN pip install --no-cache-dir build &amp;&amp; \
      cd python &amp;&amp; \
      pip wheel --no-deps --wheel-dir /wheels .

  # Stage 2: runtime -- bare alpine + python + the wheel + runner
  FROM python:3.12-alpine
  COPY --from=builder /wheels/lecerf-*.whl /tmp/
  # Mandatory size-trimming chain in the SAME RUN layer (Docker layer caching
  # would otherwise leave the deleted bytes inside the image).
  RUN pip install --no-cache-dir /tmp/lecerf-*.whl pytest \
   &amp;&amp; rm -rf /usr/local/lib/python3.12/site-packages/pip \
   &amp;&amp; rm -rf /usr/local/lib/python3.12/site-packages/setuptools \
   &amp;&amp; find /usr/local/lib/python3.12 -name __pycache__ -exec rm -rf {} + 2>/dev/null \
   &amp;&amp; find /usr/local/lib/python3.12 -name '*.pyc' -delete \
   &amp;&amp; rm -rf /tmp/*.whl /root/.cache /var/cache/apk/*
  COPY docker/runner.py /usr/local/bin/lecerf-runner
  RUN chmod +x /usr/local/bin/lecerf-runner
  ENTRYPOINT ["/usr/local/bin/lecerf-runner"]
  CMD ["--help"]

Create docker/runner.py (~50 lines):

  #!/usr/bin/env python3
  """Container entry: run pytest against firmware + tests dirs.

  Usage:
      lecerf-runner /fw /tests                  # default: junit xml on stdout
      lecerf-runner --version
      lecerf-runner --help
  """
  import os, sys, subprocess

  VERSION = "0.1.0"

  def usage():
      print(__doc__)
      sys.exit(0)

  def main(argv):
      if len(argv) == 1 or argv[1] in ("--help", "-h"):
          usage()
      if argv[1] == "--version":
          print(f"lecerf {VERSION}")
          return 0
      if len(argv) &lt; 3:
          print("error: expected /fw /tests", file=sys.stderr)
          return 2
      fw_dir, tests_dir = argv[1], argv[2]
      if not os.path.isdir(tests_dir):
          print(f"error: tests dir not found: {tests_dir}", file=sys.stderr)
          return 2
      env = dict(os.environ, LECERF_FW_DIR=fw_dir)
      cmd = ["python", "-m", "pytest", tests_dir, "-v",
             "--tb=short", "--junit-xml=/tmp/junit.xml"]
      rc = subprocess.call(cmd, env=env)
      if os.path.exists("/tmp/junit.xml"):
          # Optionally print junit to stdout for downstream consumers
          pass
      return rc

  if __name__ == "__main__":
      sys.exit(main(sys.argv))

Create docker/test_docker.sh (~50 lines, bash):

  #!/usr/bin/env bash
  # Local verification: build, size-check, run, assert pytest pass.
  set -euo pipefail
  cd "$(dirname "$0")/.."

  IMG="lecerf:dev"
  MAX_BYTES=$((50 * 1024 * 1024))   # 50 MB hard cap

  echo "=== Step 1: build image ==="
  docker build -t "$IMG" -f docker/Dockerfile .

  echo "=== Step 2: size gate ==="
  size=$(docker image inspect "$IMG" --format='{{.Size}}')
  echo "image size: $size bytes"
  if [ "$size" -gt "$MAX_BYTES" ]; then
      echo "FAIL: image exceeds 50 MB (got $size bytes)"
      exit 1
  fi
  echo "PASS: under 50 MB"

  echo "=== Step 3: --version ==="
  out=$(docker run --rm "$IMG" --version)
  echo "$out"
  echo "$out" | grep -q "lecerf 0.1.0" || { echo "FAIL version output"; exit 1; }

  echo "=== Step 4: --help ==="
  docker run --rm "$IMG" --help | grep -q "lecerf-runner"

  echo "=== Step 5: pytest run ==="
  docker run --rm \
      -v "$(pwd)/firmware:/fw:ro" \
      -v "$(pwd)/python/tests:/tests:ro" \
      "$IMG" /fw /tests

  echo "=== ALL DOCKER GATES PASSED ==="
