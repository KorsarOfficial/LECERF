  import pytest
  from lecerf import Board

  def pytest_addoption(parser):
      parser.addoption(
          "--lecerf-board",
          default="generic-m4",
          help="Default board profile for the `board` fixture",
      )

  @pytest.fixture
  def board(request):
      """Per-test Board instance. Default profile from --lecerf-board."""
      name = request.config.getoption("--lecerf-board")
      b = Board(name)
      yield b
      del b   # triggers __del__ -> board_destroy

  @pytest.fixture(params=["stm32f103", "stm32f407", "generic-m4"])
  def board_all(request):
      """Parametrized fixture: each test runs once per profile."""
      b = Board(request.param)
      yield b
      del b

Create python/tests/test_blink.py (~30 lines):

  """test_blink: GPIO + UART + register assertions for test10_stm32_blink.bin.

  This test demonstrates the three assertion classes a typical user writes:
    1. result.exit_cause -- did the firmware halt cleanly?
    2. board.uart.output() -- did UART produce expected bytes?
    3. board.gpio['GPIOC'][13].value -- is the LED pin in expected state?
  """
  import pytest
  from .conftest import fw

  @pytest.fixture
  def blink_board(board):
      """Compose: a board with blink firmware already flashed."""
      board.flash(fw("test10_stm32_blink"))
      return board

  def test_blink_halts(blink_board):
      r = blink_board.run(timeout_ms=500)
      assert r.exit_cause == "halt", f"exit_cause={r.exit_cause}"

  def test_blink_r0_counter(blink_board):
      """test10 increments R0 to 5 in its blink loop."""
      r = blink_board.run(timeout_ms=500)
      assert r.r[0] == 5, f"R0={r.r[0]}"

  def test_blink_gpio_pin(blink_board):
      """GPIOC pin 13 must be readable as a bool."""
      blink_board.run(timeout_ms=500)
      v = blink_board.gpio["GPIOC"][13].value
      assert isinstance(v, bool)

Create python/tests/test_uart.py (~25 lines):

  """test_uart: drain UART buffer and assert presence of expected bytes."""
  from .conftest import fw

  def test_uart_test3_division(board):
      """test3 computes UDIV 42/7=6 and MUL 42*7=294, prints via UART."""
      board.flash(fw("test3"))
      r = board.run(timeout_ms=200)
      assert r.exit_cause == "halt"
      assert r.r[0] == 6,    f"UDIV result R0={r.r[0]}"
      assert r.r[1] == 294,  f"MUL result R1={r.r[1]}"
      out = board.uart.output()
      assert isinstance(out, bytes)
      # test3 prints "[USART2] 'X'\n" -- not strict on exact bytes,
      # just confirm capture buffer drained successfully (idempotent re-call).
      assert board.uart.output() == b"", "second drain must be empty"

Create python/tests/test_registers.py (~40 lines):

  """test_registers: assert exact register values after running canonical firmware."""
  import pytest
  from .conftest import fw

  CASES = [
      ("test1",            "generic-m4", {0: 0x37}),
      ("test2",            "generic-m4", {0: 0x002802d0, 1: 1, 2: 9}),
      ("test4",            "generic-m4", {0: 5}),
      ("test10_stm32_blink", "stm32f103", {0: 5}),
      ("test11_nvic_irq",  "generic-m4", {0: 1, 1: 10}),
  ]

  @pytest.mark.parametrize("fwname,profile,expected", CASES,
                           ids=[c[0] for c in CASES])
  def test_register_state(fwname, profile, expected):
      from lecerf import Board
      b = Board(profile)
      r = b.flash(fw(fwname)).run(timeout_ms=2000)
      assert r.exit_cause in ("halt", "timeout"), (
          f"{fwname}: exit_cause={r.exit_cause}")
      for reg, val in expected.items():
          assert r.r[reg] == val, f"{fwname}: R{reg}={r.r[reg]:#x} expected {val:#x}"
